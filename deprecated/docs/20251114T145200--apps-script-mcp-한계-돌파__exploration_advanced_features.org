*** MCP의 한계

#+begin_src yaml
MCP 할 수 있는 것:
  ✅ API 호출 (Gmail, Calendar, Drive 등)

  ❌ HTML UI (복잡한 인터페이스)

// =PROJECT_HOURS("SKS Gateway", "2025-11-11", "2025-11-15")
function PROJECT_HOURS(projectName, startDate, endDate) {
  var cal = CalendarApp.getDefaultCalendar();
  var start = new Date(startDate);
  var end = new Date(endDate);
  var events = cal.getEvents(start, end);

  var hours = 0;
  events.forEach(function(e) {
    if (e.getTitle().includes(projectName)) {
      hours += (e.getEndTime() - e.getStartTime()) / (1000 * 60 * 60);
    }
  });

  return hours;
}

// =HOURLY_RATE(B2, C2)  // hours, revenue
function HOURLY_RATE(hours, revenue) {
  if (hours === 0) return 0;
  return revenue / hours;
}

// 2. Custom Functions (Sheets 셀에서)
function PROJECT_HOURS(name, days) {
  var cal = CalendarApp.getDefaultCalendar();
  var events = cal.getEvents(
    new Date(Date.now() - days * 24 * 60 * 60 * 1000),
    new Date()
  );

  return events
    .filter(e => e.getTitle().includes(name))
    .reduce((sum, e) =>
      sum + (e.getEndTime() - e.getStartTime()) / 3600000, 0
    );
}

function HOURLY_RATE(hours, revenue) {
  return hours > 0 ? revenue / hours : 0;
}

function DEADLINE_STATUS(deadline) {
  var days = (new Date(deadline) - new Date()) / (1000 * 60 * 60 * 24);

  if (days < 0) return "⛔ 지연";
  if (days < 3) return "🔥 긴급";
  if (days < 7) return "⚠️ 주의";
  return "✅ 여유";
}

// 3. Event Trigger (Progress 업데이트 시)
function onEdit(e) {
  var sheet = e.source.getActiveSheet();
  if (sheet.getName() !== 'Projects') return;

  var range = e.range;
  var col = range.getColumn();

  // Progress 열 (5번)
  if (col === 5) {
    var row = range.getRow();
    var progress = range.getValue();
    var projectName = sheet.getRange(row, 1).getValue();

    // 100% 완료 시
    if (progress >= 100) {
      // Chat 알림
      sendChatMessage('team-space',
        '🎉 ' + projectName + ' 프로젝트 완료!');

      // 자동 아카이브 제안
      var ui = SpreadsheetApp.getUi();
      var response = ui.alert(
        '프로젝트 완료',
        projectName + '을(를) 아카이브할까요?',
        ui.ButtonSet.YES_NO
      );

      if (response === ui.Button.YES) {
        archiveProject(row);
      }
    }
  }
}

// 4. Time-driven (매일 9시)
function dailyDeadlineCheck() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName('Projects');
  var data = sheet.getDataRange().getValues();

  var urgent = [];

  data.forEach(function(row, i) {
    if (i === 0) return;  // 헤더

    var deadline = new Date(row[3]);  // Deadline 열
    var days = (deadline - new Date()) / (1000 * 60 * 60 * 24);

    if (days < 3 && days >= 0) {
      urgent.push({
        name: row[0],
        days: Math.ceil(days)
      });
    }
  });

  if (urgent.length > 0) {
    var message = '⚠️ 마감 임박:\n' +
      urgent.map(u => '- ' + u.name + ' (' + u.days + '일)').join('\n');

    // Chat + Gmail 알림
    sendChatMessage('team-space', message);
    GmailApp.sendEmail('user@example.com', '마감 임박 알림', message);
  }
}

** 💡 핵심 인사이트

*** Apps Script = Workspace의 백엔드

#+begin_quote
"MCP는 API 클라이언트다.
Apps Script는 Workspace의 백엔드다.

MCP: 요청 → 응답
Apps Script: 상주 → 자동 → UI → 통합

둘을 결합하면:
Claude가 필요할 때만 개입하고,
나머지는 Apps Script가 자율 처리한다."
#+end_quote

*** Custom Functions = Sheets의 두뇌

#+begin_quote
"Custom Functions로 Sheets를 살아있는 대시보드로 만든다.

=PROJECT_HOURS("SKS", 7)
=HOURLY_RATE(E2, F2)
=DEADLINE_STATUS(C2)

Sheets를 열면 모든 계산이 자동으로 완료되어 있다.
Claude는 그 결과만 읽으면 된다 (토큰 절약)."
#+end_quote

*** UI Components = 사람의 인터페이스

#+begin_quote
"Claude는 코드로 협업하지만,
일반인은 UI로 협업한다.

Sidebar, Menu, Dialog로
팀원들도 에이전트 시스템을 사용할 수 있다."
#+end_quote

** 🎓 결론

#+begin_quote
"Apps Script는 MCP의 한계를 넘어선다.

MCP가 할 수 없는 것:
- UI (Menus, Sidebars, Dialogs)
- Custom Functions (Sheets 셀 함수)
- Event-driven (onEdit, onFormSubmit)
- Advanced Services (깊은 API 제어)
- Libraries (코드 재사용)

이 모든 것이 Claude가 Workspace를
'사용'하는 것을 넘어
'확장'하게 만든다.

Apps Script = Workspace의 백엔드
MCP = Workspace의 API 클라이언트
Claude = 오케스트레이터

셋의 조합이 진정한 에이전트 시스템이다."
